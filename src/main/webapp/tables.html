Taulukko 1. Väkiluku ikäryhmittäin 1. tammikuuta ja ennuste

SELECT ?ik ?v sum(?val) WHERE { ?i scv:dimension ?v . ?v rdf:type dimension:Vuosi . FILTER ( ?v = vuosi:_2000 || ?v = vuosi:_2008 || ?v = vuosi:_2009 || ?v = vuosi:_2010 ) ?i scv:dimension yksikkö:Henkilöä . ?i scv:dimension ?ik . ?ik rdf:type dimension:Ikäryhmä . ?i scv:dimension ?a . ?a rdf:type dimension:Alue . OPTIONAL { ?a skos:broader ?broader } . FILTER ( !BOUND(?broader) ) ?i rdf:value ?val . } GROUP BY ?v ?ik ORDER BY ?ik ?v
